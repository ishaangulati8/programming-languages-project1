<term>		-> <factor> { (MULT_OP | DIV_OP | MOD_OP) <factor> }

<factor>	-> NUM [<exponent>]
			-> OPEN_PAREN <expr> CLOSE_PAREN [<exponent>]

<exponent>	-> POW_OP <expr> [<exponent>]
